# Button

A flexible button component available as plain CSS classes, React components, and Astro components.

## Usage

<Demo code={`<button class="zui-button">Click me</button>
<a href="/action" class="zui-button">Link as button</a>`}>
  <button class="zui-button">Click me</button>
  <a href="/action" class="zui-button">Link as button</a>
</Demo>

## Button Types

Text-only, text with icon, and icon-only buttons at each size.

{['xs', 'sm', 'md', 'lg', 'xl'].map((size) => {
  const sizeClass = size !== 'md' ? ` zui-button-size-${size}` : ''
  return (
    <ButtonRow label={size}>
      <button class={`zui-button${sizeClass}`}>Button</button>
      <button class={`zui-button${sizeClass}`}><i class="ph ph-plus"></i> With icon</button>
      <button class={`zui-button zui-button-icon${sizeClass}`} aria-label="Add"><i class="ph ph-plus"></i></button>
    </ButtonRow>
  )
})}

## Variants × Accents

{['fill', 'subtle', 'outline', 'ghost', 'link'].map((variant) => (
  <ButtonRow label={variant}>
    {['primary', 'secondary', 'destructive'].map((accent) => (
      <button class={`zui-button zui-button-variant-${variant} zui-button-accent-${accent}`}>
        {accent}
      </button>
    ))}
  </ButtonRow>
))}

## Sizes × Shapes

{['default', 'hard', 'soft', 'squircle'].map((shape) => (
  <ButtonRow label={shape}>
    {['xs', 'sm', 'md', 'lg', 'xl'].map((size) => {
      const classes = [
        'zui-button',
        'zui-button-accent-secondary',
        size !== 'md' ? `zui-button-size-${size}` : '',
        shape !== 'default' ? `zui-button-shape-${shape}` : '',
      ].filter(Boolean).join(' ')
      return <button class={classes}>{size}<i class="ph ph-arrow-right"></i></button>
    })}
  </ButtonRow>
))}
