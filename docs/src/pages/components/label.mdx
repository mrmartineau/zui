# Label

Form labels with optional required indicator and description text.

## Usage

<Demo code={`<label class="zui-label" for="demo">Email address</label>
<input class="zui-input" type="email" id="demo" placeholder="you@example.com" />`}>
  <div style="width: 100%;">
    <label class="zui-label" for="demo">Email address</label>
    <input class="zui-input" type="email" id="demo" placeholder="you@example.com" />
  </div>
</Demo>

<CodeTabs
  html={`<label class="zui-label" for="email">Email address</label>
<input class="zui-input" type="email" id="email" />`}
  react={`import { Label, Input } from '@mrmartineau/zui/react'

<Label htmlFor="email">Email address</Label>
<Input type="email" id="email" />`}
  astro={`---
import { Label, Input } from '@mrmartineau/zui/astro'
---

<Label for="email">Email address</Label>
<Input type="email" id="email" />`}
/>
