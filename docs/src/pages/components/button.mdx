# Button

A flexible button component available as plain CSS classes, React components, and Astro components.

## Usage

<Demo
  html={`<button class="zui-button">Click me</button>
<a href="/action" class="zui-button">Link as button</a>`}
  react={`import { Button } from '@mrmartineau/zui/react'

<Button>Click me</Button>
<Button href="/action">Link as button</Button>`}
  astro={`---
import { Button } from '@mrmartineau/zui/astro'
---

<Button>Click me</Button>
<Button href="/action">Link as button</Button>`}
  solid={`import { Button } from '@mrmartineau/zui/solid'

<Button>Click me</Button>
<Button href="/action">Link as button</Button>`}
  vue={`<template>
  <Button>Click me</Button>
  <Button href="/action">Link as button</Button>
</template>

<script setup>
import { Button } from '@mrmartineau/zui/vue'
</script>`}
>
  <button class="zui-button">Click me</button>
  <a href="/action" class="zui-button">Link as button</a>
</Demo>

## Variants × Accents

{['fill', 'subtle', 'outline', 'ghost', 'link'].map((variant) => (
  <ButtonRow label={variant}>
    {['theme', 'accent', 'destructive'].map((color) => (
      <button class={`zui-button zui-button-variant-${variant} zui-button-color-${color}`}>
        {color}
      </button>
    ))}
  </ButtonRow>
))}

## Sizes × Shapes

{['default', 'hard', 'soft', 'squircle'].map((shape) => (
  <ButtonRow label={shape}>
    {['xs', 'sm', 'md', 'lg', 'xl'].map((size) => {
      const classes = [
        'zui-button',
        'zui-button-color-accent',
        size !== 'md' ? `zui-button-size-${size}` : '',
        shape !== 'default' ? `zui-button-shape-${shape}` : '',
      ].filter(Boolean).join(' ')
      return <button class={classes}>{size}<i class="ph ph-arrow-right"></i></button>
    })}
  </ButtonRow>
))}
