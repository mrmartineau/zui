# Input

A styled text input for forms.

## Usage

<Demo code={`<input class="zui-input" type="text" placeholder="Enter your name" />`}>
  <input class="zui-input" type="text" placeholder="Enter your name" />
</Demo>

<CodeTabs
  html={`<input class="zui-input" type="text" placeholder="Enter your name" />`}
  react={`import { Input } from '@mrmartineau/zui/react'

<Input type="text" placeholder="Enter your name" />`}
  astro={`---
import { Input } from '@mrmartineau/zui/astro'
---

<Input type="text" placeholder="Enter your name" />`}
  solid={`import { Input } from '@mrmartineau/zui/solid'

<Input type="text" placeholder="Enter your name" />`}
  vue={`<script setup>
import { Input } from '@mrmartineau/zui/vue'
</script>

  <Input type="text" placeholder="Enter your name" />
</template>`}
/>
