## Usage

<Demo
  html={`<div class="flex gap-md">
  <div>One</div>
  <div>Two</div>
  <div>Three</div>
</div>`}
  react={`import { Flex } from '@mrmartineau/zui/react'

<Flex gap="md">
  <div>One</div>
  <div>Two</div>
  <div>Three</div>
</Flex>`}
  astro={`---
import { Flex } from '@mrmartineau/zui/astro'
---

<Flex gap="md">
  <div>One</div>
  <div>Two</div>
  <div>Three</div>
</Flex>`}
  solid={`import { Flex } from '@mrmartineau/zui/solid'

<Flex gap="md">
  <div>One</div>
  <div>Two</div>
  <div>Three</div>
</Flex>`}
  vue={`<template>
  <Flex gap="md">
    <div>One</div>
    <div>Two</div>
    <div>Three</div>
  </Flex>
</template>

<script setup>
import { Flex } from '@mrmartineau/zui/vue'
</script>`}
>
  <div class="flex gap-md">
    <div class="zui-badge">One</div>
    <div class="zui-badge">Two</div>
    <div class="zui-badge">Three</div>
  </div>
</Demo>

## Props

| Prop | Values | Default |
|------|--------|---------|
| `display` | `flex`, `inline` | `flex` |
| `direction` | `row`, `rowReverse`, `column`, `columnReverse` | — |
| `align` | `start`, `end`, `center`, `baseline`, `stretch` | — |
| `justify` | `start`, `end`, `center`, `between`, `around` | — |
| `wrap` | `wrap`, `nowrap`, `wrapReverse` | — |
| `gap` | `0`, `2xs`, `xs`, `sm`, `md`, `lg`, `xl`, `2xl`, `3xl`, `4xl` | — |
| `gapX` | `0`, `2xs`, `xs`, `sm`, `md`, `lg`, `xl`, `2xl`, `3xl`, `4xl` | — |
| `gapY` | `0`, `2xs`, `xs`, `sm`, `md`, `lg`, `xl`, `2xl`, `3xl`, `4xl` | — |
