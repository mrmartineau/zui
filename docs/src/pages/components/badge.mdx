## All Variants × Colours

{['subtle', 'fill', 'outline'].map((variant) => {
  const variantClass = variant !== 'subtle' ? ` zui-badge-variant-${variant}` : ''
  return (
    <>
      <h3>{variant}</h3>
      <Demo html={`<!-- ${variant} variant -->`}>
        {['red', 'orange', 'amber', 'yellow', 'lime', 'green', 'emerald', 'teal', 'cyan', 'sky', 'blue', 'indigo', 'violet', 'purple', 'fuchsia', 'pink', 'rose', 'gray'].map((color) => (
          <span class={`zui-badge${variantClass} zui-badge-color-${color}`}>{color}</span>
        ))}
      </Demo>
    </>
  )
})}
